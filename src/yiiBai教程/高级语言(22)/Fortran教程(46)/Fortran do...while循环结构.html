Fortran do...while循环结构

它重复语句或一组语句，当给定的条件为真。它测试的条件执行在循环体之前。

语法

do while (logical expr) 
   statements
end do

流程图

示例

program factorial  
implicit none  

   ! define variables
   integer :: nfact = 1   
   integer :: n = 1 
   
   ! compute factorials   
   do while (n <= 10)       
      nfact = nfact * n 
      n = n + 1
      print*,  n, " ", nfact   
   end do 
end program factorial
